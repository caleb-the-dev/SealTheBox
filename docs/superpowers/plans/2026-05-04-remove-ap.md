# Remove AP (Action Points) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Strip Action Points entirely from the game — no AP variable, no spend_ap(), no ap_cost field, no AP UI, no AP mentions in docs.

**Architecture:** AP was a per-round resource (reset to 3 each round) that gated die rolls (1 AP each) and was referenced on ability cards. Removing it means rolling dice is now free (no resource cost). Abilities remain charge-based; only the unused ap_cost field is removed from AbilityData. The CSV column shift (ap_cost was col 5; description/charges shift left) must be applied atomically with the loader change.

**Tech Stack:** GDScript (Godot 4.5), CSV data, Markdown docs

- [ ] **Step 4: Edit ability_library.gd — remove ap_cost parsing, fix column indices**

Replace the entire file with:

```gdscript
extends Node

var _abilities: Dictionary = {}

func _ready() -> void:
	_load_csv()

func _load_csv() -> void:
	var file = FileAccess.open("res://data/abilities.csv", FileAccess.READ)
	if not file:
		push_error("AbilityLibrary: cannot open res://data/abilities.csv")
		return
	file.get_csv_line()  # skip header row
	while not file.eof_reached():
		var row = file.get_csv_line()
		if row.size() < 6 or row[0].strip_edges().is_empty():
			continue
		if not row[4].is_valid_int():
			push_warning("AbilityLibrary: skipping malformed row (non-integer cooldown): %s" % row[0])
			continue
		var data = AbilityData.new()
		data.id = row[0].strip_edges()
		data.flavor_name = row[1].strip_edges()
		data.type = row[2].strip_edges()
		var traits_raw = row[3].strip_edges()
		data.traits.clear()
		if traits_raw != "":
			for t in traits_raw.split(",", false):
				data.traits.append(t.strip_edges())
		data.cooldown = row[4].to_int()
		data.description = row[5].strip_edges()
		data.charges = max(1, row[6].to_int()) if row.size() > 6 and row[6].strip_edges().is_valid_int() else 1
		data.max_charges = data.charges
		_abilities[data.id] = data
	file.close()

func get_ability(id: String) -> AbilityData:
	return _abilities.get(id, null)

func get_all() -> Array:
	return _abilities.values()
```

```gdscript
func _init() -> void:
```

```gdscript
func _test_draw_hand() -> void:
	var pool = DicePool.new()
	pool.setup([Die.new(6), Die.new(6), Die.new(6), Die.new(8)])
	var hand = pool.draw_hand()
	assert(hand.size() == 3, "Hand should have 3 dice, got %d" % hand.size())

func _test_roll() -> void:
	var pool = DicePool.new()
	pool.setup([Die.new(6), Die.new(6), Die.new(6), Die.new(8)])
	var hand = pool.draw_hand()

	assert(die.rolled, "Die should be marked rolled")
	assert(val >= 1 and val <= die.faces, "Rolled value %d out of range [1,%d]" % [val, die.faces])
	assert(die.value == val, "die.value should match returned value")

func _test_modifiers() -> void:
	var pool = DicePool.new()
	pool.setup([Die.new(6)])
	var hand = pool.draw_hand()
	var die = hand[0]
	die.value = 4
	die.rolled = true
	pool.apply_greater(die, 1)
	assert(die.value == 5, "Greater 1: 4+1=5, got %d" % die.value)
	pool.apply_greater(die, 10)
	assert(die.value == 6, "Greater capped at face max 6, got %d" % die.value)
	pool.apply_lesser(die, 2)
	assert(die.value == 4, "Lesser 2: 6-2=4, got %d" % die.value)
	die.value = 1
	pool.apply_lesser(die, 5)
	assert(die.value == 1, "Lesser capped at 1, got %d" % die.value)
	for i in 20:
		pool.reroll(die)
		assert(die.value >= 1 and die.value <= die.faces, "Reroll out of range: %d" % die.value)

func _test_discard_and_reshuffle() -> void:
	var pool = DicePool.new()
	pool.setup([Die.new(6), Die.new(6), Die.new(6), Die.new(8)])
	var hand = pool.draw_hand()
	assert(pool.get_hand().size() == 3, "Hand size 3 after draw")
	pool.discard_hand()
	assert(pool.get_hand().size() == 0, "Hand cleared after discard")
	var hand2 = pool.draw_hand()
	assert(hand2.size() == 3, "After reshuffle, draw 3 again")
```

```gdscript
func start_round() -> void:
	GameState.round += 1
	if GameState.round > GameState.round_limit:
		GameState.hp -= 1
```

```gdscript
func commit_roll(dice: Array) -> void:
	for die in dice:
		_dice_pool.roll_die(die)
	var total := 0
	for die in GameState.dice_hand:
		if die.rolled:
			total += die.value
	_set_phase("act")
	status_updated.emit("Seal Phase — Total: %d — select tabs that sum to it." % total)
```
